create table EMPLOYES (
EMPLOYEE_ID NUMBER(0) IS PRIMARY KEY NOT NULL,
FIRST_NAME VARCHAR2(20),
LAST_NAME VARCHAR2(20),
EMAIL VARCHAR2(20) IS UNIQUE,
PHONE_NUMBER VARCHAR2(20),
HIRE_DATE DATE,
JOB_ID VARCHAR2(10),
SALARY NUMBER(6,2),
COMMISSION_PCT NUMBER(2,2),
MANAGER_ID NUMBER(0),
DEPARTAMENT_ID NUMBER(4)
);

create index NONULO on EMPLOYES (SALARY)
WHERE SALARY IS NOT NULL  ;
